<script>
const optionsPool = [
    { label: "+10分", value: 10, img: "https://images.plurk.com/6oPvP96Ve3Db7vAo43h4i8.png" }, 
    { label: "+30分", value: 30, img: "https://images.plurk.com/3EeeBcQv2o8FbmgDXUpZzk.png" },
    { label: "+50分", value: 50, img: "https://images.plurk.com/33tJ55TLelKAbqWS6px51y.png" }, 
    { label: "+2時", value: 120, img: "https://images.plurk.com/5WjPCqbBJIXVoxJLf30U9Q.png" }, 
    { label: "+4時", value: 240, img: "https://images.plurk.com/7lzeGRy3WrI01WQsYnibJW.png" },
    { label: "+9時", value: 540, img: "https://images.plurk.com/5TCFiPZEXPd3jCCvcXIVsD.png" }, 
    { label: "x 2", value: "multiply", img: "https://images.plurk.com/5DBC7BsTBST7hDMJhhF7v7.png" }
];

let currentMinutes = 0, initialMinutes = 0, currentButtons = [];
const TARGET = 710, LIMIT = 720;

function initGame() {
    initialMinutes = (Math.floor(Math.random() * 67) * 10) + 30;
    currentMinutes = initialMinutes;
    currentButtons = [];
    for (let i = 0; i < 4; i++) {
        currentButtons.push({ ...optionsPool[Math.floor(Math.random() * optionsPool.length)] });
    }
    updateTimeDisplay();
    renderButtons();
    const msg = document.getElementById('message');
    msg.innerText = "請點選按鈕或結束控時";
    msg.style.color = "#c9d1d9";
    document.getElementById('best-solution-container').style.display = "none";
    document.getElementById('next-btn').style.display = "none";
    document.getElementById('end-btn').style.display = "block";
}

function renderButtons() {
    const container = document.getElementById('button-container');
    container.innerHTML = '';
    currentButtons.forEach((opt, index) => {
        const btn = document.createElement('button');
        const img = document.createElement('img');
        img.src = opt.img;
        img.alt = opt.label;
        btn.appendChild(img);
        btn.onclick = () => applyEffect(index, btn);
        container.appendChild(btn);
    });
}

function updateTimeDisplay() {
    const h = Math.floor(currentMinutes / 60);
    const m = currentMinutes % 60;
    document.getElementById('current-time').innerText = 
        `${h.toString().padStart(2, '0')}:${m.toString().padStart(2, '0')}`;
}

function applyEffect(index, btnElement) {
    const opt = currentButtons[index];
    if (opt.value === "multiply") currentMinutes *= 2;
    else currentMinutes += opt.value;
    btnElement.disabled = true;
    updateTimeDisplay();
    if (currentMinutes >= LIMIT || currentMinutes === TARGET) finishGame();
}

function solveBest() {
    let bestVal = -1, bestCombo = [];
    for (let i = 0; i < 16; i++) {
        let tempMin = initialMinutes, tempLabels = [];
        for (let j = 0; j < 4; j++) {
            if ((i >> j) & 1) {
                let opt = currentButtons[j];
                if (opt.value === "multiply") tempMin *= 2;
                else tempMin += opt.value;
                tempLabels.push(opt.label);
            }
        }
        if (tempMin < LIMIT && tempMin > bestVal) {
            bestVal = tempMin; bestCombo = tempLabels;
        }
    }
    return { val: bestVal, combo: bestCombo };
}

function finishGame() {
    const msg = document.getElementById('message'), best = solveBest();
    document.getElementById('best-solution-container').style.display = "block";
    const bh = Math.floor(best.val / 60), bm = best.val % 60;
    const bestTimeStr = `${bh.toString().padStart(2, '0')}:${bm.toString().padStart(2, '0')}`;
    if (currentMinutes === TARGET || (currentMinutes < LIMIT && currentMinutes === best.val)) {
        msg.innerText = "★ 完美控時 ★"; msg.style.color = "#2ecc71";
    } else if (currentMinutes >= LIMIT) {
        msg.innerText = "超過 12:00，失敗！"; msg.style.color = "#ff4757";
    } else {
        msg.innerText = "未達最佳解，再接再厲"; msg.style.color = "#f1c40f";
    }
    document.getElementById('best-solution').innerText = 
        `本題最佳解：${bestTimeStr}\n建議步驟：${best.combo.join(' -> ') || '不需控時'}`;
    document.querySelectorAll('#button-container button').forEach(b => b.disabled = true);
    document.getElementById('end-btn').style.display = "none";
    document.getElementById('next-btn').style.display = "block";
}

document.getElementById('end-btn').onclick = finishGame;
document.getElementById('next-btn').onclick = initGame;
initGame();
